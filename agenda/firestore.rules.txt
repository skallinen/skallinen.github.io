// ============================================================
// COMPLETE Firestore rules: existing bookrank (clubs) rules
// + the agenda app (agendas/**), in the same style.
// REPLACE the whole rules document in Firebase Console →
// Firestore → Rules with this file.
//
// Agenda conventions follow the club conventions exactly:
// - membership = existence of members/{uid} subdoc
// - any signed-in user can read top-level docs (invite-code
//   lookup query needs it) and create their own membership
//   (the invite code is a lookup convenience, not a secret
//   the rules can verify — same PoC trade-off as clubs)
// - superusers retain full access
// ============================================================

rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    // Helper: check if user is a member of the club
    function isMember(clubId) {
      return exists(/databases/$(database)/documents/clubs/$(clubId)/members/$(request.auth.uid));
    }
    // Helper: check if user is the club creator
    function isCreator(clubId) {
      return get(/databases/$(database)/documents/clubs/$(clubId)).data.created_by == request.auth.uid;
    }
    // Helper: check if user is a superuser
    function isSuperuser() {
      return exists(/databases/$(database)/documents/superusers/$(request.auth.uid));
    }
    // Helper: check if user is a member of the agenda
    function isAgendaMember(agendaId) {
      return exists(/databases/$(database)/documents/agendas/$(agendaId)/members/$(request.auth.uid));
    }
    // Helper: check if user is the agenda creator
    function isAgendaCreator(agendaId) {
      return get(/databases/$(database)/documents/agendas/$(agendaId)).data.created_by == request.auth.uid;
    }

    // Superusers collection: only superusers can read; no client writes
    // (grant via Firebase Console or admin script)
    match /superusers/{userId} {
      allow read: if request.auth != null && request.auth.uid == userId;
      allow write: if false; // admin-only, managed via console/scripts
    }

    // ---------------- BOOKRANK (unchanged) ----------------
    match /clubs/{clubId} {
      // Any authenticated user can read clubs (needed for join-by-invite-code query)
      allow read: if request.auth != null;
      // Any authenticated user can create a new club
      allow create: if request.auth != null;
      // Members or superusers can update
      allow update: if request.auth != null && (isMember(clubId) || isSuperuser());
      // Only the creator or superuser can delete the club
      allow delete: if request.auth != null && (isCreator(clubId) || isSuperuser());
      // Members
      match /members/{memberId} {
        allow read: if request.auth != null;
        allow create, update: if request.auth != null
          && memberId == request.auth.uid;
        allow delete: if request.auth != null
          && (memberId == request.auth.uid || isCreator(clubId) || isSuperuser());
      }
      // Books: club members or superusers
      match /books/{bookId} {
        allow read: if request.auth != null && (isMember(clubId) || isSuperuser());
        allow create: if request.auth != null && (isMember(clubId) || isSuperuser());
        allow update, delete: if request.auth != null && (isMember(clubId) || isSuperuser());
      }
      // Rankings: members can read; users write own; creator/superuser can delete any
      match /rankings/{userId} {
        allow read: if request.auth != null && (isMember(clubId) || isSuperuser());
        allow create, update: if request.auth != null
          && userId == request.auth.uid
          && isMember(clubId);
        allow delete: if request.auth != null
          && (userId == request.auth.uid || isCreator(clubId) || isSuperuser());
      }
    }

    // ---------------- AGENDA (new) ----------------
    match /agendas/{agendaId} {
      // Any authenticated user can read agendas (needed for join-by-invite-code query)
      allow read: if request.auth != null;
      // Any authenticated user can create a new agenda
      allow create: if request.auth != null;
      // Members or superusers can update (join flow: the member doc is
      // created first, so the subsequent member_uids update passes)
      allow update: if request.auth != null && (isAgendaMember(agendaId) || isSuperuser());
      // Only the creator or superuser can delete the agenda
      allow delete: if request.auth != null && (isAgendaCreator(agendaId) || isSuperuser());

      // Members: you may create/update your own membership (join);
      // creator/superuser may remove anyone
      match /members/{memberId} {
        allow read: if request.auth != null;
        allow create, update: if request.auth != null
          && memberId == request.auth.uid;
        allow delete: if request.auth != null
          && (memberId == request.auth.uid || isAgendaCreator(agendaId) || isSuperuser());
      }

      // Planning data — agenda members or superusers.
      // Explicit per collection (no wildcard, so the members rules
      // above stay the only path to member docs).
      match /persons/{docId} {
        allow read, write: if request.auth != null && (isAgendaMember(agendaId) || isSuperuser());
      }
      match /periods/{docId} {
        allow read, write: if request.auth != null && (isAgendaMember(agendaId) || isSuperuser());
      }
      match /marks/{docId} {
        allow read, write: if request.auth != null && (isAgendaMember(agendaId) || isSuperuser());
      }
      match /anchors/{docId} {
        allow read, write: if request.auth != null && (isAgendaMember(agendaId) || isSuperuser());
      }
      match /notes/{docId} {
        allow read, write: if request.auth != null && (isAgendaMember(agendaId) || isSuperuser());
      }
    }
  }
}
